\section{Ciclo de Refrigeração Padrão:}

\begin{equation}
    Q_L = \dot{m}(h_1-h_4)
    \label{QL}
\end{equation}

\begin{equation}
    Q_H = \dot{m}(h_2-h_3)
    \label{QH}
\end{equation}

\begin{equation}
    W_{comp} = \dot{m}(h_2-h_1)
    \label{W compressor}
\end{equation}

\begin{equation}
    \Delta h_{1-2} \simeq  c_p (T_2-T_1)
    \label{simplificacao entalpia}
\end{equation}

\begin{equation}
    COP = \frac{T_H}{T_H - T_L}
    \label{COP carnot}
\end{equation}
